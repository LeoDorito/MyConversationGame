#!/usr/bin/env python3 """ LeoToWEB.py - Convert Leo.mylang to Leo_WEB.html Completely rewritten with proper block tracking """ import re import os def convert_mylang_to_web(): with open('Leo.mylang', 'r', encoding='utf-8') as f: lines = f.readlines() js_game_code = generate_js_game_code(lines) html_content = generate_html(js_game_code) with open('Leo_WEB.html', 'w', encoding='utf-8') as f: f.write(html_content) os.makedirs('website_upload', exist_ok=True) with open('website_upload/index.html', 'w', encoding='utf-8') as f: f.write(html_content) print("SUCCESS! Created Leo_WEB.html") print("Also updated website_upload/index.html for Netlify!") def escape_js(text): """Escape text for JavaScript string""" return text.replace('\\', '\\\\').replace("'", "\\'").replace('\n', '\\n') def generate_js_game_code(lines): """Convert Leo.mylang lines to JavaScript game code""" js_lines = [] indent_level = 0 # Track how deep we are in blocks def indent(): return ' ' + (' ' * indent_level) for line in lines: stripped = line.strip() # Skip empty lines and comments if not stripped or stripped.startswith('Comment:'): continue # Handle End FIRST (before other commands) to properly close blocks if stripped == 'End' or stripped == 'End ': if indent_level > 0: indent_level -= 1 js_lines.append(f"{indent()}}}") continue # Write command starting with string: Write "text" if stripped.startswith('Write "') and stripped.endswith('"'): text = stripped[7:-1] # Handle string concatenation like "text" + var + "more" if '" + ' in text: parts = text.split('" + ') result = f"'{escape_js(parts[0])}'" for part in parts[1:]: if part.endswith(' + "'): var = part[:-4] result += f" + {var} + " elif ' + "' in part: idx = part.index(' + "') var = part[:idx] rest = part[idx+4:] result += f" + {var} + '{escape_js(rest)}'" else: result += f" + {part}" js_lines.append(f"{indent()}this.writeText({result});") else: js_lines.append(f"{indent()}this.writeText('{escape_js(text)}');") # Write command starting with variable: Write var + "text" or Write var +"text" elif stripped.startswith('Write ') and ('+' in stripped) and not stripped.startswith('Write "'): # Normalize: add spaces around + if missing content = stripped[6:].replace('+', ' + ') # Remove double spaces while ' ' in content: content = content.replace(' ', ' ') parts = content.split(' + ') result = "" for i, part in enumerate(parts): part = part.strip() if part.startswith('"') and part.endswith('"'): result += f"'{escape_js(part[1:-1])}'" elif part.startswith('"'): result += f"'{escape_js(part[1:])}'" elif part.endswith('"'): result += f"'{escape_js(part[:-1])}'" else: result += part if i < len(parts) - 1: result += " + " js_lines.append(f"{indent()}this.writeText({result});") # WriteRaw command elif stripped.startswith('WriteRaw "') and stripped.endswith('"'): text = stripped[10:-1] js_lines.append(f"{indent()}this.writeText('{escape_js(text)}');") # Ask command elif stripped.startswith('Ask "') and stripped.endswith('"'): text = stripped[5:-1] js_lines.append(f"{indent()}this.lastAnswer = await this.askQuestion('{escape_js(text)}');") # Save command elif stripped.startswith('Save ') and ' = ' in stripped: match = re.search(r'Save (\w+) = (.+)', stripped) if match: var_name = match.group(1) value = match.group(2).strip() if value == 'answer': js_lines.append(f"{indent()}var {var_name} = this.lastAnswer;") else: js_lines.append(f"{indent()}var {var_name} = '{value}';") # Choose command elif stripped.startswith('Choose "'): parts = re.findall(r'"([^"]*)"', stripped) if len(parts) >= 2: question = escape_js(parts[0]) options = [escape_js(opt) for opt in parts[1:]] options_str = ', '.join([f"'{opt}'" for opt in options]) js_lines.append(f"{indent()}this.lastAnswer = await this.showChoices('{question}', [{options_str}]);") # Random command elif stripped.startswith('Random '): match = re.search(r'Random (\d+) (\d+)', stripped) if match: min_val = match.group(1) max_val = match.group(2) js_lines.append(f"{indent()}this.lastAnswer = String(Math.floor(Math.random() * ({max_val} - {min_val} + 1)) + {min_val});") # Beep command elif stripped.startswith('Beep '): match = re.search(r'Beep (\d+) (\d+)', stripped) if match: freq = match.group(1) duration = match.group(2) js_lines.append(f"{indent()}await this.beep({freq}, {duration});") # Exit command elif stripped == 'Exit': js_lines.append(f"{indent()}return; // Exit the game") # Repeat command elif stripped.startswith('Repeat ') and 'times:' in stripped: match = re.search(r'Repeat (\d+) times:', stripped) if match: count = min(int(match.group(1)), 100) js_lines.append(f"{indent()}for (var _i = 0; _i < {count}; _i++) {{") indent_level += 1 # If equals command with quoted string elif stripped.startswith('If ') and ' equals "' in stripped and stripped.endswith(':'): match = re.search(r'If (\w+) equals "([^"]+)":', stripped) if match: var_name = match.group(1) value = match.group(2) js_lines.append(f"{indent()}if (String({var_name}) === '{escape_js(value)}') {{") indent_level += 1 # If equals command with variable (no quotes) elif stripped.startswith('If ') and ' equals ' in stripped and stripped.endswith(':') and '"' not in stripped.split(' equals ')[1]: match = re.search(r'If (\w+) equals (\w+):', stripped) if match: var_name = match.group(1) other_var = match.group(2) js_lines.append(f"{indent()}if (String({var_name}) === String({other_var})) {{") indent_level += 1 # If greater than command (supports both variables and quoted strings) elif stripped.startswith('If ') and ' greater than ' in stripped and stripped.endswith(':'): match = re.search(r'If (\w+) greater than ["\']?([^"\':\s]+)["\']?:', stripped) if not match: match = re.search(r'If (\w+) greater than "([^"]+)":', stripped) if match: var1 = match.group(1) var2 = match.group(2) js_lines.append(f"{indent()}if (parseInt({var1}) > parseInt('{var2}')) {{") indent_level += 1 # If less than command (supports both variables and quoted strings) elif stripped.startswith('If ') and ' less than ' in stripped and stripped.endswith(':'): match = re.search(r'If (\w+) less than ["\']?([^"\':\s]+)["\']?:', stripped) if not match: match = re.search(r'If (\w+) less than "([^"]+)":', stripped) if match: var1 = match.group(1) var2 = match.group(2) js_lines.append(f"{indent()}if (parseInt({var1}) < parseInt('{var2}')) {{") indent_level += 1 return '\n'.join(js_lines) def generate_html(js_game_code): return ''' The Conversation Game! :D
